# 第四题
建立S表：
```SQL
S(SNO, SNAME, STATUS, CITY);
CREATE TABLE S
(SNO CHAR(3),
SNAME CHAR(3),
STATUS CHAR(2),
CITY CHAR(10));
```
建立P表：
```SQL
P(PNO, PNAME, COLOR, WEIGHT);
CREATE TABLE P
(PNO CHAR(3),
PNAME CHAR(10),
COLOR CHAR(4),
WEIGHT INT;
```
建立J表：
```SQL
J(JNO, JNAME, CITY);
CREATE TABLE J
(JNO CHAR(3),
JNAME CHAR(3),
CITY CHAR(10));
```
建立SPJ表：
```SQL
SPJ(SNO, PNO, JNO, QTY);
CREATE TABLE SPJ
(SNO CHAR(3),
PNO CHAR(3),
JNO CHAR(3),
QTY INT);
```
求供应J1的SNO：
```SQL
SELECT SNO
FROM SPJ
WHERE JNO = "J1";
```
求供应J1的P1零件的SNO：
```SQL
SELECT SNO
FROM SPJ
WHERE JNO = "J1" AND PNO = "P1"
```
求J1 红色的SNO：
```SQL
SELECT SNO
FROM SPJ
WHERE JNO = "J1"
AND PNO IN
	(SELECT PNO
	FROM P
	WHERE COLOR = "红");
```
求没有使用天津的红色零件的JNO：
```SQL
SELECT JNO
FROM J
WHERE NO EXISTS
	(SELECT *
	FROM SPJ, S, P
	WHERE SPJ.JNO = J.JNO AND SPJ.SNO = S.SNO ANDSPJ.PNO = P.PNO AND S.CITY = "天津" AND P.COLOR = "红")
```
求至少使用S1的全部零件的JNO：
```SQL
SELECT DISTINCT JNO
FROM SPJ SPJZ
WHERE NO EXISTS
	(SELECT *
	FROM SPJ SPJX
	WHERE SNO = "S1"
	AND NOT EXISTS
		(SELECT *
		FROM SPJ SPJY
		WHERE SPJY.PNO = SPJX.PNO AND SPJY.JNO = SPJZ.JNO));
```
# 第五题
找出所有供应商的姓名和所在城市
```SQL
SELECT SNAME, CITY
FROM S;
```
找出所有零件的名称、颜色、重量
```SQL
SELECT PNAME, COLOR, WEIGHT
FROM P;
```
找出使用供应商S1所供应零件的工程号码
```SQL
SELECT JNO
FROM SPJ
WHERE SNO = "S1";
```
找出工程项目J2使用的各种零件的名称及其数量
```SQL
SELECT P.PNAME, SPJ.QTY
FROM P, SOJ
WHERE P.PNO = SPJ.PNO AND SPJ.JNO = "J2"
```
找出上海厂商供应的所有零件号码
```SQL
SELECT DISTINCT PNO
FROM SPJ
WHERE SNO IN
	(SELECT SNO
	FROM S
	WHERE CITY = "上海");
```
找出使用上海产的零件的工程名称
```SQL
SELECT JNAME
FROM J, SPJ, S
WHERE J.JNO = SPJ.JNO AND SPJ.SNO = S.SNO AND S.CITY = "上海";
```
找出没有使用天津产的零件的工程号码
```SQL
SELECT JNO
FROM J
WHERE NOT EXISTS
	(SELECT *
	FROM SPJ
	WHERE SPJ.JNO = J.JNO
	AND SNO IN
		(SELECT SNO
		FROM S
		WHERE CITY = "天津")); 
```
把全部红色零件的颜色改成蓝色
```SQL
UPDATE P
SET COLOR = "蓝"
WHERE COLOR = "红";
```
由S5供给J4的零件P6改为由S3供应
```SQL
UPDATE SPJ
SET SNO = "S5" AND JNO = "J4" AND PNO = "P6";
```
从供应商关系中删除S2的记录，并从供应情况中删除相应的记录
```SQL
DELETE
FROM SPJ
WHERE SNO = "S2";

FROM S

```
